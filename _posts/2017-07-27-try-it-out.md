function is_prime(x) {
    var y;
    if (x < 2) return false;
    if (x == 2) return true;
    if (x % 2 == 0) return false;
    for (y = 3; y * y <= x; y += 2) {
        if (x % y == 0) {
            return false;
        }
    }
    return true;
}
